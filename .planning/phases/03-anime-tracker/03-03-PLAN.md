   export async function getStaticPaths() {
     const items = await getAllAnime();
     const slugs = items.map((a) => slugOf(a.data.titleCn));
     if (new Set(slugs).size !== slugs.length) {
       const dupes = slugs.filter((s, i) => slugs.indexOf(s) !== i);
       throw new Error(`[anime] duplicate slug: ${[...new Set(dupes)].join(', ')}`);

     return items.map((a) => ({
       params: { slug: slugOf(a.data.titleCn) },
       props: { entry: a },
     }));
   }
